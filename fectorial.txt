DECLARE
    n       NUMBER := &enter_number;   
    result  NUMBER := 1;
BEGIN
    IF n < 0 THEN
        DBMS_OUTPUT.PUT_LINE('Factorial is not defined for negative numbers.');
    ELSIF n = 0 THEN
        DBMS_OUTPUT.PUT_LINE('Factorial of 0 is 1');
    ELSE
        FOR i IN 1..n LOOP
            result := result * i;
        END LOOP;

        DBMS_OUTPUT.PUT_LINE('Factorial of ' || n || ' is ' || result);
    END IF;
END;
/
